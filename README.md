## 1. 프로젝트 구성

```
├── pom.xml                                                   [ maven repository dependency & build config ]
└── src                                                       [ application source 디렉토리 ]
    ├── main                                                  [ source main 디렉토리 ( common ) ]
    │   ├── java                                              [ java source 디렉토리 ]
    │   │   └── common                                        [ common source 디렉토리 ]
    │   │       ├── component                                 [ common에 사용될 component 디렉토리 ]
    │   │       │   └── PaginationInfo.java                   [ pagenation common util ]
    │   │       ├── dao                                       [ common dao 디렉토리 ]
    │   │       │   ├── AbstractDAO.java                      [ common dao 추상 dao ]
    │   │       │   └── CommonDAO.java                        [ common 사용할 쿼리 ( 파일관련등 ) ]
    │   │       ├── dto                                       [ common model 디렉토리 ]
    │   │       │   └── FileVO.java                           [ common 파일처리관련 model( POJO ) ]
    │   │       ├── interceptor                               [ common interceptor 디렉토리 ]
    │   │       │   └── ExecuteTimeInterceptor.java           [ common interceptor ( 쿼리시간측정, 트래픽관련 등 ) ]
    │   │       ├── map                                       [ common request 파라미터 바인딩 & 쿼리 resultset 바인딩 디렉토리 ]
    │   │       │   ├── CommandMap.java                       [ request 파라미터 바인딩 ]
    │   │       │   └── QueryMap.java                         [ 데이터베이스 쿼리 resultset 바인딩 ]
    │   │       ├── resolver                                  [ resolver 디렉토리 ]
    │   │       │   └── CustomMapArgumentResolver.java        [ request resolver( resolver재정의 ) commandMap 바인딩 ]
    │   │       ├── sql                                       [ common sql 디렉토리 ]
    │   │       │   └── Common_SQL.xml                        [ common sql 정의( 페이징 등 ) ]
    │   │       └── util                                      [ common util 디렉토리 ]
    │   │           ├── BaseCompressUtil.java                 [ 압축, 압축풀기 util ]
    │   │           ├── BaseFileUtil.java                     [ 파일 업로드 & 확장자 체크 관련 util ]
    │   │           ├── CamelUtil.java                        [ from snake case to camel case util ]
    │   │           ├── DownloadUtil.java                     [ 파일 다운로드 util ]
    │   │           ├── NumberUtil.java                       [ number 관련 util ]
    │   │           ├── RandomStringBuilder.java              [ randomstring util ]
    │   │           ├── StringUtil.java                       [ string util ( null & "" 등 ) ]
    │   │           ├── UuidUtil.java                         [ uuid util( 파일네이밍처리 등 ) ]
    │   │           ├── WebUtil.java                          [ web util ]
    │   │           └── XmlUtil.java                          [ xml parsing 관련 util ]
    │   ├── resources                                         [ config resource 디렉토리 ]
    │   │   ├── common                                        [ common resource 디렉토리 ]
    │   │   │   ├── context                                   [ common context 환경 디렉토리 ]
    │   │   │   │   ├── context-common.xml                    [ common spring bean 주입설정 ( DI ) ]
    │   │   │   │   └── context-datasource.xml                [ common database bean 주입설정 ( DI ) ]
    │   │   │   └── prop                                      [ common 설정 property 디렉토리 ]
    │   │   │       └── global.properties                     [ common 설정 property 정의 ]
    │   │   └── log4j.xml                                     [ common log 관련 ]
    │   ├── skeletone                                         [ 실제 application project 디렉토리 현재명은 ( skeletone ) ]
    │   │   ├── java                                          [ application java source 디렉토리 ]
    │   │   │   └── skeletone                                 [ applicaotion 디렉토리 ]
    │   │   │       ├── attach                                [ application 파일처리 관련 디렉토리 ]
    │   │   │       │   ├── AttachedFileDAO.java              [ application 파일처리 관련 db dao ]
    │   │   │       │   ├── AttachedFileVO.java               [ application 파일처리 관련 model ( POJO ) ]
    │   │   │       │   └── AttachedFile_SQL.xml              [ application 파일처리 관련 sql ]
    │   │   │       ├── interceptor                           [ application interceptor 디렉토리 ]
    │   │   │       │   └── LoginCheckInterceptor.java        [ application login 세션 확인 interceptor ]
    │   │   │       ├── main                                  [ application main 디렉토리 ]
    │   │   │       │   └── MainController.java               [ application main controller ]
    │   │   │       ├── member                                [ application member 디렉토리 ]
    │   │   │       │   ├── MemberController.java             [ application member controller ]
    │   │   │       │   ├── MemberDAO.java                    [ application member db dao ]
    │   │   │       │   ├── MemberService.java                [ application member service interface]
    │   │   │       │   ├── MemberServiceImpl.java            [ application member interface 구현체 ]
    │   │   │       │   ├── MemberVO.java                     [ application member model ( POJO ) ]
    │   │   │       │   └── Member_SQL.xml                    [ application member sql ]
    │   │   │       ├── sample                                [ application member sample 디렉토리 ]
    │   │   │       │   ├── SampleController.java             [ application sample controller ]
    │   │   │       │   ├── SampleDAO.java                    [ application sample db dao ]
    │   │   │       │   ├── SampleService.java                [ application sample service interface ]
    │   │   │       │   ├── SampleServiceImpl.java            [ application sample interface 구현체 ]
    │   │   │       │   └── Sample_SQL.xml                    [ application sample sql ]
    │   │   │       └── util                                  [ application util 디렉토리 ]
    │   │   │           └── MessageUtil.java                  [ application messageutil ]
    │   │   └── resources                                     [ application config resource 디렉토리 ]
    │   │       └── skeletone                                 [ application 환경설정 디렉토리 ]
    │   │           ├── config                                [ application 설정 디렉토리 ]
    │   │           │   ├── MapperConfig.xml                  [ application mybatis 설정 ]
    │   │           │   └── jobConfig.properties              [ application 설정 정의 ]
    │   │           └── context                               [ application 환경설정 ]
    │   │               └── context-job.xml                   [ application spring bean 주입설정 ( DI ) ]
    │   └── webapp                                            [ web sources root 디렉토리 ]
    │       ├── WEB-INF                                       [ WEB-INF ]
    │       │   ├── spring                                    [ spring ]
    │       │   │   └── appServlet                            [ application servlet 설정 디렉토리 ]
    │       │   │       └── servlet-context.xml               [ application servlet 설정 정의 ]
    │       │   ├── views                                     [ application web resources 디렉토리 ]
    │       │   │   ├── alert                                 [ application 공통 alert 디렉토리 ]
    │       │   │   │   ├── logoutAlert.jsp                   [ application 로그아웃 처리 alert, redirect]
    │       │   │   │   └── notLoginAlert.jsp                 [ application 로그인 되지않음 처리 alert, redirect  ]
    │       │   │   ├── main                                  [ application 메인 디렉토리 ]
    │       │   │   │   └── index.jsp                         [ application main index 페이지 ]
    │       │   │   └── member                                [ application 회원관련 디렉토리 ]
    │       │   │       └── login.jsp                         [ application 로그인 페이지 ]
    │       │   └── web.xml                                   [ application servlet, filter, context path 정의 web.xml ]
    │       ├── resources                                     [ application static 접근 디렉토리 ]
    │       │   └── js                                        [ application javascript 디렉토리 ]
    │       │       ├── bx_ichicsr.js                         [ bx_ichicsr.js ]
    │       │       ├── default.js                            [ default.js ]
    │       │       ├── handlebars.min.js                     [ handlebars.min.js ]
    │       │       ├── jquery.min.js                         [ jquery.min.js ]
    │       │       ├── jquery.uploadify.min.js               [ jquery.uploadify.min.js ]
    │       │       ├── moment.min.js                         [ moment.min.js ]
    │       │       ├── overview.js                           [ overview.js ]
    │       │       ├── spin.js                               [ spin.js ]
    │       │       └── temp_validation.js                    [ temp_validation.js ]
    │       └── skeletone                                     [ application 명 디렉토리 ( 현재 예시 skeletone ) ]
    │           ├── main                                      [ application 메인 디렉토리 ]
    │           │   └── index.jsp                             [ application 메인 redirect ]
    │           └── member                                    [ application 회원 디렉토리 ]   
    │               ├── login                                 [ application 로그인 디렉토리 ]
    │               │   └── index.jsp                         [ application 로그인 redirect ]
    │               └── logout                                [ application 로그아웃 디렉토리 ]
    │                   └── index.jsp                         [ application 로그아웃 redirect ]
    └── test                                                  [ application test 디렉토리 ]
        ├── resources                                         [ application test resources 디렉토리 ]
        │   └── log4j.xml                                     [ application test log4j ]
        └── skeletone                                         [ application test 디렉토리]
            └── java                                          [ application test java 디렉토리 ]
                └── skeletone                                 [ application test java ]
                    └── SampleControllerTest.java             [ application test source ]

```
